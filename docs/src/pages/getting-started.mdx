## Installation

```shell
npm install primer-react-prototype
```

```shell
yarn add primer-react-prototype
```

## ThemeProvider

```jsx
import React from "react"
import { ThemeProvider } from "primer-react-protoype"

function App() {
  return (
    <ThemeProvider>
      <div>...</div>
    </ThemeProvider>
  )
}
```
